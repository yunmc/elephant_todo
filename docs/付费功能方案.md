## 一、付费方案总览

### 定价模型：Freemium（免费 + 高级订阅）

| 项目 | 说明 |
|------|------|
| **免费版** | 全部基础功能不限制使用 |
| **高级版 (Premium)** | 解锁多账本、记录附图、AI 功能（快速记账/报告） |
| **手帐商店（象币）** | 独立于订阅，用象币购买皮肤/贴纸/字体/宠物皮肤，自由搭配 |
| **定价** | ¥3/月 或 ¥28/年（首年优惠 ¥18） |
| **支付方式** | 初期走 Web 端微信/支付宝；上架后考虑 Apple IAP |

### 免费 vs 付费功能划分

| 模块 | 免费版 | 高级版 (Premium) |
|------|--------|-----------------|
| 待办管理 | ✅ 完整功能 | — |
| 随手记 | ✅ 完整功能 | — |
| 密码本 | ✅ 完整功能 | — |
| 重要日期 | ✅ 完整功能 | — |
| 经期追踪 | ✅ 完整功能 | — |
| 记账 | ✅ 基础记账（单账本、人民币） | 🔓 多账本 + 预算管理 + 多币种 |
| 外观 | ✅ 亮色/暗色 | 🏪 手帐商店：用象币购买皮肤/贴纸/字体/宠物皮肤（独立于订阅） |
| AI 宠物 | ✅ 每日一言 + 情绪互动 | 🏪 象币喂食解锁 AI 聊天/智能提醒 |
| 记录附图 | ❌ | 🔓 图片上传 |
| AI 功能 | ❌ | 🔓 智能匹配 + AI 快速记账 + AI 月度/年度报告 |

```typescript
// composables/usePremium.ts
export function usePremium() {
  const authStore = useAuthStore()

  const isPremium = computed(() => {
    if (!authStore.user) return false
    if (authStore.user.plan !== 'premium') return false
    if (!authStore.user.plan_expires_at) return false
    return new Date(authStore.user.plan_expires_at) > new Date()
  })

  /** 非会员时引导升级 */
  function guardPremium(): boolean {
    if (isPremium.value) return true
    // 弹出升级引导弹窗
    showUpgradeModal()
    return false
  }

  return { isPremium, guardPremium }
}
```

### 4.7 核心逻辑

**购买流程**：

```typescript
// server/api/shop/purchase.post.ts
async function purchaseProduct(userId: number, productId: number) {
  const product = await getProduct(productId)

  // 检查是否已拥有
  const owned = await isProductOwned(userId, productId)
  if (owned) throw createError({ statusCode: 400, message: '已拥有该商品' })

  // 检查限时商品时效
  if (product.is_limited) {
    const now = new Date()
    if (now < product.limited_start || now > product.limited_end)
      throw createError({ statusCode: 400, message: '该商品已下架' })
  }

  // 事务：扣象币 + 解锁商品
  await db.transaction(async (tx) => {
    // 1. 扣减余额（乐观锁）
    const result = await tx.execute(
      'UPDATE user_wallets SET balance = balance - ?, total_spent = total_spent + ? WHERE user_id = ? AND balance >= ?',
      [product.price, product.price, userId, product.price]
    )
    if (result.affectedRows === 0)
      throw createError({ statusCode: 400, message: '象币余额不足' })

    // 2. 记录流水
    const wallet = await getWallet(tx, userId)
    await tx.execute(
      'INSERT INTO wallet_transactions (user_id, type, amount, balance_after, reference_type, reference_id, description) VALUES (?, "purchase", ?, ?, "product", ?, ?)',
      [userId, -product.price, wallet.balance, productId, `购买${product.name}`]
    )

    // 3. 解锁商品
    if (product.type === 'bundle') {
      // 套装：解锁所有包含的单品
      const items = await getBundleItems(tx, productId)
      for (const item of items) {
        await tx.execute(
          'INSERT IGNORE INTO user_products (user_id, product_id, source) VALUES (?, ?, "bundle")',
          [userId, item.product_id]
        )
      }
    }
    // 同时记录套装/单品本身
    await tx.execute(
      'INSERT IGNORE INTO user_products (user_id, product_id, source) VALUES (?, ?, "purchase")',
      [userId, productId]
    )
  })
}
```

```typescript
  // 从服务器加载当前装扮
  async function loadAppearance() {
    const data = await useApi('/api/user/appearance')
    appearance.value = data
    applyToDOM(data)
  }

  // 将装扮应用到页面
  function applyToDOM(config: UserAppearance) {
    if (!import.meta.client) return
    const root = document.documentElement

    // 皮肤：设置 CSS 变量 + 背景
    if (config.skin) {
      root.setAttribute('data-skin', config.skin.asset_key)
    } else {
      root.removeAttribute('data-skin')
    }

    // 字体：加载 Web Font
    if (config.font) {
      loadWebFont(config.font.asset_key)
      root.style.setProperty('--font-family', config.font.asset_key)
    }

    // 贴纸：通过 CSS class 控制显示
    if (config.stickerPack) {
      root.setAttribute('data-stickers', config.stickerPack.asset_key)
    }
  }

  // 更新装扮（自由搭配）
  async function updateAppearance(config: Partial<UserAppearance>) {
    await useApi('/api/user/appearance', {
      method: 'PUT',
      body: {
        skin_id: config.skin?.id ?? appearance.value.skin?.id ?? null,
        sticker_pack_id: config.stickerPack?.id ?? appearance.value.stickerPack?.id ?? null,
        font_id: config.font?.id ?? appearance.value.font?.id ?? null
      }
    })
    await loadAppearance()
  }

  return { appearance, loadAppearance, updateAppearance }
}
```

### 5.5 前端实现

```typescript
// composables/useAttachment.ts
export function useAttachment() {
  const { isPremium, guardPremium } = usePremium()

  async function upload(file: File, targetType: string, targetId: number) {
    if (!guardPremium()) return null

    // 前端压缩
    const compressed = await compressImage(file, { maxWidth: 1080, quality: 0.8 })

    const formData = new FormData()
    formData.append('file', compressed)
    formData.append('target_type', targetType)
    formData.append('target_id', String(targetId))

    const res = await $fetch('/api/attachments/upload', {
      method: 'POST',
      body: formData,
    })
    return res.data
  }

  return { upload }
}
```

```typescript
// server/api/pet/feed.post.ts
async function feedPet(userId: number) {
  const FEED_COST = 5          // 象币/次（待定）
  const STAMINA_GAIN = 20      // 恢复体力

  await db.transaction(async (tx) => {
    // 1. 扣象币
    const result = await tx.execute(
      'UPDATE user_wallets SET balance = balance - ?, total_spent = total_spent + ? WHERE user_id = ? AND balance >= ?',
      [FEED_COST, FEED_COST, userId, FEED_COST]
    )
    if (result.affectedRows === 0)
      throw createError({ statusCode: 400, message: '象币不足，快去充值吧～' })

    // 2. 恢复体力（不超上限）
    await tx.execute(
      'UPDATE user_pets SET stamina = LEAST(stamina + ?, max_stamina), mood = "happy", updated_at = NOW() WHERE user_id = ?',
      [STAMINA_GAIN, userId]
    )

    // 3. 记录流水 & 喂食日志
    const wallet = await getWallet(tx, userId)
    await tx.execute(
      'INSERT INTO wallet_transactions (user_id, type, amount, balance_after, reference_type, description) VALUES (?, "purchase", ?, ?, "pet_feed", "喂食小象")',
      [userId, -FEED_COST, wallet.balance]
    )
    await tx.execute(
      'INSERT INTO pet_feed_log (user_id, cost, stamina_gained) VALUES (?, ?, ?)',
      [userId, FEED_COST, STAMINA_GAIN]
    )
  })
}
```

```typescript
// server/api/pet/chat.post.ts
async function petChat(userId: number, message: string) {
  const STAMINA_COST = 5

  // 1. 检查体力
  const pet = await getPet(userId)
  if (pet.stamina < STAMINA_COST)
    throw createError({ statusCode: 400, message: '小象饿了，先喂食吧～🍎' })

  // 2. 扣体力
  await db.execute(
    'UPDATE user_pets SET stamina = stamina - ?, total_chats = total_chats + 1, updated_at = NOW() WHERE user_id = ?',
    [STAMINA_COST, userId]
  )

  // 3. 获取最近聊天记录作为上下文
  const history = await getRecentChats(userId, 20)

  // 4. 构建 Prompt
  const personalityPrompts = {
    cute: '你是一只名叫{name}的可爱小象，性格软萌温柔，说话喜欢用颜文字和语气词。',
    sarcastic: '你是一只名叫{name}的傲娇小象，说话喜欢吐槽但内心关心主人，偶尔傲娇。',
    encouraging: '你是一只名叫{name}的元气小象，性格阳光积极，总是鼓励和夸奖主人。'
  }

  const systemPrompt = personalityPrompts[pet.personality]
    .replace('{name}', pet.name)
    + '\n你是 Elephant App 的 AI 宠物助手。回复简短亲切（50字以内），可以适当使用 emoji。'

  // 5. 调用 Deepseek（便宜）
  const reply = await callLLM({
    model: 'deepseek',
    system: systemPrompt,
    messages: [
      ...history.map(h => ({ role: h.role === 'pet' ? 'assistant' : 'user', content: h.content })),
      { role: 'user', content: message }
    ],
    maxTokens: 150
  })

  // 6. 保存聊天记录
  await saveChat(userId, 'user', message, 0)
  await saveChat(userId, 'pet', reply, STAMINA_COST)

  await updateMood(userId)

  return { reply, stamina: pet.stamina - STAMINA_COST }
}
```

### 12.1 Premium 订阅过期降级策略

| 功能 | 过期后处理 | 说明 |
|------|-----------|------|
| **多账本** | 只读，不可新增/编辑记录 | 数据不删除，续费后恢复可写 |
| **预算管理** | 只读，不可新建/修改预算 | 历史预算数据保留 |
| **多币种** | 只读，新记录只能用人民币 | 已有多币种记录保留可查看 |
| **记录附图** | 只读，不可新增图片 | 已上传图片保留可查看 |
| **AI 智能匹配** | 不可用，显示升级引导 | — |
| **AI 快速记账** | 不可用，显示升级引导 | — |
| **AI 月度/年度报告** | 历史报告可查看，不可生成新报告 | 已缓存的报告保留 |

> 核心原则：**数据永不删除，只锁写不锁读**。用户续费后立即恢复全部功能。

**实现方式**：

```typescript
// server/utils/premium.ts
export function requirePremium(event: H3Event) {
  const user = getAuthUser(event)
  if (!user.plan_expires_at || new Date(user.plan_expires_at) < new Date()) {
    throw createError({
      statusCode: 403,
      message: 'PREMIUM_REQUIRED',
      data: { expired: !!user.plan_expires_at } // 区分"从未订阅"和"已过期"
    })
  }
}

// 前端捕获
function handlePremiumError(error: any) {
  if (error.data?.message === 'PREMIUM_REQUIRED') {
    if (error.data?.data?.expired) {
      showRenewModal() // "您的会员已过期，续费后可继续使用"
    } else {
      showUpgradeModal() // "升级 Premium 解锁此功能"
    }
  }
}
```
